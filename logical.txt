import java.io.*;
class Relational {
    public static void main(String[] args) {
        int a = 1, b = 2;

        System.out.println("a is " + a + " and b is " + b);

        System.out.println("a == b: " + (a == b));
        System.out.println("a != 6: " + (a != 6));
        System.out.println("a > b: " + (a > b));
        System.out.println("a < b: " + (a < b));
        System.out.println("a >= b: " + (a >= b));
        System.out.println("a <= b: " + (a <= b));
    }
}
